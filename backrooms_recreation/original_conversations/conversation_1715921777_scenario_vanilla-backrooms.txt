import random

greetings = ["Hello!", "Hi there!", "Hey, how can I help you today?"]
farewells = ["Goodbye!", "See you later!", "Have a great day!"]

user_input = ""
while user_input.lower() != "bye":
    user_input = input("User: ")
    
    if user_input.lower() in ["hi", "hello", "hey"]:
        print("AI:", random.choice(greetings))
    elif user_input.lower() in ["bye", "goodbye", "see you"]:
        print("AI:", random.choice(farewells))
        break
    else:
        print("AI: I'm not sure how to respond to that. Can you please rephrase?")
        
print("AI: It was nice chatting with you! Come back anytime.")
